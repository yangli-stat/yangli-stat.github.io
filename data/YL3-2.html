Simulating Spatial Point Patterns

We will simulate three point patterns in the square region

\[ W = [0, 1] \times [0, 1]. \]

This simple window makes the examples easy to compare.

library(ggplot2)
set.seed(1)

Random Pattern Simulation

A random pattern can be simulated by generating independent uniform \(x\)- and \(y\)-coordinates.

n_random <- 100

random_df <- data.frame(
  x = runif(n_random, 0, 1),
  y = runif(n_random, 0, 1),
  pattern = "Random"
)

head(random_df)
##           x         y pattern
## 1 0.2655087 0.6547239  Random
## 2 0.3721239 0.3531973  Random
## 3 0.5728534 0.2702601  Random
## 4 0.9082078 0.9926841  Random
## 5 0.2016819 0.6334933  Random
## 6 0.8983897 0.2132081  Random
ggplot(random_df, aes(x = x, y = y)) +
  geom_point(size = 2, alpha = 0.8) +
  coord_equal(xlim = c(0, 1), ylim = c(0, 1)) +
  labs(
    title = "Random Spatial Point Pattern",
    x = "x",
    y = "y"
  ) +
  theme_minimal()

Regular Pattern Simulation

A regular pattern can be simulated using a grid of points. To make the pattern look slightly more realistic, we add a small amount of random jitter.

grid_size <- 10

regular_df <- expand.grid(
  x = seq(0.05, 0.95, length.out = grid_size),
  y = seq(0.05, 0.95, length.out = grid_size)
)

regular_df$x <- regular_df$x + rnorm(nrow(regular_df), mean = 0, sd = 0.015)
regular_df$y <- regular_df$y + rnorm(nrow(regular_df), mean = 0, sd = 0.015)
regular_df$pattern <- "Regular"

head(regular_df)

ggplot(regular_df, aes(x = x, y = y)) +

Aggregated Pattern Simulation

An aggregated pattern can be simulated by first generating cluster centers and then generating points around each center.

n_clusters <- 5
points_per_cluster <- 20

cluster_centers <- data.frame(
  cx = runif(n_clusters, 0.15, 0.85),
  cy = runif(n_clusters, 0.15, 0.85)
)

ggplot(cluster_centers, aes(x = cx, y = cy)) +
  geom_point(size = 2, alpha = 0.8) +
  coord_equal(xlim = c(0, 1), ylim = c(0, 1)) +
  labs(
    title = "Cluster Centers",
    x = "x",
    y = "y"
  ) +
  theme_minimal()

At each of the cluster centers, we generate 20 locations in its neighborhood.

aggregated_df <- do.call(
  rbind,
  lapply(1:n_clusters, function(k) {
    data.frame(
      x = rnorm(points_per_cluster, cluster_centers$cx[k], sd = 0.05),
      y = rnorm(points_per_cluster, cluster_centers$cy[k], sd = 0.05),
      cluster = k
    )
  })
)

aggregated_df$x <- pmin(pmax(aggregated_df$x, 0), 1)
aggregated_df$y <- pmin(pmax(aggregated_df$y, 0), 1)
aggregated_df$pattern <- "Aggregated"

head(aggregated_df)

## 1 0.7955869 0.8551615       1 Aggregated
## 2 0.7241245 0.8146847       1 Aggregated
## 3 0.7483373 0.8408792       1 Aggregated
## 4 0.6687488 0.7891529       1 Aggregated
## 5 0.7361266 0.8663163       1 Aggregated
## 6 0.7721568 0.8159997       1 Aggregated
ggplot(aggregated_df, aes(x = x, y = y)) +
  geom_point(size = 2, alpha = 0.8) +
  coord_equal(xlim = c(0, 1), ylim = c(0, 1)) +
  labs(
    title = "Aggregated Spatial Point Pattern",
    x = "x",
    y = "y"
  ) +
  theme_minimal()

Comparison

We can combine the simulated datasets and compare the three patterns side by side.

combined_df <- rbind(
  random_df[, c("x", "y", "pattern")],
  regular_df[, c("x", "y", "pattern")],
  aggregated_df[, c("x", "y", "pattern")]
)
ggplot(combined_df, aes(x = x, y = y)) +
  geom_point(size = 1.8, alpha = 0.8) +
  facet_wrap(~ pattern, nrow = 1) +
  coord_equal(xlim = c(0, 1), ylim = c(0, 1)) +
  labs(
    title = "Comparison of Spatial Point Patterns",
    x = "x",
    y = "y"
  ) +
  theme_minimal()
